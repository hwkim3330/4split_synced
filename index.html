Quick Start

# Clone repository
git clone https://github.com/hwkim3330/4split_synced.git
cd 4split_synced

# Auto sync based on filename timestamps
python3 4split_sync.py video1.mp4 video2.mp4 video3.mp4 video4.mp4 --auto-sync -o output.mp4

# Manual offsets (in seconds)
python3 4split_sync.py video1.mp4 video2.mp4 video3.mp4 video4.mp4 --offsets 0 3 1 1 -o output.mp4
